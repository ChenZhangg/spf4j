17  /**
18   * mojo that packages a schema package and avro sources and attaches them as separate artifacts.
19   */
20  @Mojo(name = "avro-package", defaultPhase = LifecyclePhase.PACKAGE, requiresProject = true)
21  public final class SchemaPackageMojo extends SchemaMojoBase {
22  
23    public String[] getSourceFiles() {
24      FileSetManager fsm = new FileSetManager();
25      FileSet fs = new FileSet();
26      fs.setDirectory(sourceDirectory.getAbsolutePath());
27      fs.addInclude("**/*.avsc");
28      fs.addInclude("**/*.avpr");
29      fs.addInclude("**/*.avdl");
30      fs.setFollowSymlinks(false);
31      return fsm.getIncludedFiles(fs);
32    }
33  
34    /**
35     * {@inheritDoc} running packaging of the current project may package a script for execution Dependencies libraries
36     * are copied in the
37     *
38     *
39     */
40    public void execute() throws MojoExecutionException {
41      Log logger = this.getLog();
42      logger.info("Packaging schemas");
43      Path avsc = target.toPath().resolve(
44                mavenProject.getArtifactId() + '-' + mavenProject.getVersion() + '-' + "avsc.jar");
45      try {
46        Compress.zip(this.generatedAvscTarget.toPath(), avsc);
47      } catch (IOException ex) {
48        throw new MojoExecutionException("Cannot package schemas and sources from " + this.generatedAvscTarget, ex);
49      }
50      DefaultArtifact schemas = new DefaultArtifact(mavenProject.getGroupId(),
51              mavenProject.getArtifactId(), mavenProject.getVersion(), "compile", "jar", "avsc",
52              new DefaultArtifactHandler("jar"));
53      schemas.setFile(avsc.toFile());
54      mavenProject.getAttachedArtifacts().add(schemas);
55      Path sources = target.toPath().resolve(
56                mavenProject.getArtifactId() + '-' + mavenProject.getVersion() + '-' + "avroSources.jar");
57      try {
58        Compress.zip(this.sourceDirectory.toPath(), sources);
59      } catch (IOException ex) {
60        throw new MojoExecutionException("Cannot package sources from " + this.sourceDirectory, ex);
61      }
62      DefaultArtifact avroSources = new DefaultArtifact(mavenProject.getGroupId(),
63              mavenProject.getArtifactId(), mavenProject.getVersion(), "compile", "jar", "avroSources",
64              new DefaultArtifactHandler("jar"));
65      avroSources.setFile(sources.toFile());
66      mavenProject.getAttachedArtifacts().add(avroSources);
67    }
