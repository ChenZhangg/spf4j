102   private Spf4jTestLogRunListenerSingleton() {
103     minLogLevel = Level.valueOf(System.getProperty("spf4j.test.log.collectMinLevel", "DEBUG"));
104     maxDebugLogsCollected = Integer.getInteger("spf4j.test.log.collectmaxLogs", 100);
105     defaultTestTimeoutMillis =  Long.getLong("spf4j.test.log.defaultTestTimeoutMillis", 120000);
106     collectPrinted = Boolean.getBoolean("spf4j.test.log.collectPrintedLogs");
107     excludeLogsFromCollection = Csv.readSystemProperty("spf4j.test.log.collectExclusions");
108     baggages = new ConcurrentHashMap<>();
109     synchronized (Thread.class) {
110       final Thread.UncaughtExceptionHandler defaultHandler = Thread.getDefaultUncaughtExceptionHandler();
111       uncaughtExceptionHandler = new ExceptionAsserterUncaughtExceptionHandler(defaultHandler);
112       Thread.setDefaultUncaughtExceptionHandler(uncaughtExceptionHandler);
113     }
114   }

120   @Override
121   public synchronized void testRunFinished(final Result result) {
122     List<UncaughtExceptionDetail> exceptions = uncaughtExceptionHandler.getUncaughtExceptions();
123     if (!exceptions.isEmpty()) {
124       AssertionError assertionError = new AssertionError("Uncaught exceptions encountered " + exceptions);
125       for (UncaughtExceptionDetail ex : exceptions) {
126         Throwable throwable = ex.getThrowable();
127         assertionError.addSuppressed(throwable);
128         LOG.info("Uncaught exceptions, failures = {} in thread {}", result.getFailures(),
129                 ex.getThread(), throwable);
130       }
131       throw assertionError;
132     }
133   }

135   private void dumpDebugInfo(final Collection<TestLogRecord> logs,
136           final Description description) {
137     Iterator<TestLogRecord> iterator = logs.iterator();
138     if (iterator.hasNext()) {
139       synchronized (System.out) { // do not interleave other stuff.
140         LogPrinter.PRINTER.printTo(System.out, new TestLogRecordImpl("test", Level.INFO,
141                 "Dumping last {} logs collected for debug for {}", maxDebugLogsCollected, description), "");
142         TestLogRecord record = iterator.next();
143         LogPrinter.PRINTER.printTo(System.out, record, "");
144         while (iterator.hasNext()) {
145           record = iterator.next();
146           LogPrinter.PRINTER.printTo(System.out, record, "");
147         }
148         LogPrinter.PRINTER.printTo(System.out, new TestLogRecordImpl("test", Level.INFO,
149                 "End debug log dump for {}", description), "");
150       }
151     }

155   private static void registerDeadlockLogger(final Description description,
156           final ExecutionContext ctx, final long delay, final TimeUnit tu) {
157     final long delayMillis = Math.max(tu.toMillis(delay) - 10,  0);
158     ScheduledFuture<?> future = SCHEDULER.schedule(() -> {
159       LOG.info("Unit test  {} did not finish after {} {}, dumping thread stacks", description, delay, tu);
160       Threads.dumpToPrintStream(System.err);
161     }, delayMillis, TimeUnit.MILLISECONDS);
162     ctx.addCloseable(() -> {
163           future.cancel(true);
164         });
165   }

167   @Override
168   @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
169   public synchronized void testStarted(final Description description) throws Exception {
170     Test ta = description.getAnnotation(Test.class);
171     ExecutionContext ctx;
172     if (TestUtils.isExecutedWithDebuggerAgent()) {
173       ctx = ExecutionContexts.start(description.getDisplayName());
174     } else if (ta != null && ta.timeout() > 0) {
175       ctx = ExecutionContexts.start(description.getDisplayName(), ta.timeout(), TimeUnit.MILLISECONDS);
176       registerDeadlockLogger(description, ctx, ta.timeout(), TimeUnit.MILLISECONDS);
177     } else {
178       ctx = ExecutionContexts.start(description.getDisplayName(), defaultTestTimeoutMillis, TimeUnit.MILLISECONDS);
179       registerDeadlockLogger(description, ctx, defaultTestTimeoutMillis, TimeUnit.MILLISECONDS);
180     }
181     TestLoggers sysTest = TestLoggers.sys();
182     LogCollection<ArrayDeque<TestLogRecord>> collectLogs = handleLogCollections(description, sysTest);
183     List<LogAssert> logExpectations = setUpUnitTestLogExpectations(description, sysTest);
184     baggages.put(description, new TestBaggage(ctx, collectLogs, logExpectations));
185     handlePrintLogAnnotations(description, sysTest);
186     super.testStarted(description);
187   }
188 
189     private List<LogAssert> setUpUnitTestLogExpectations(final Description description,
190           final TestLoggers sysTest) {
191       List<LogAssert> assertions = new ArrayList<>(2);
192       assertions.add(
193                   sysTest.dontExpect("", Level.ERROR,
194                   Matchers.allOf(LogMatchers.noAttachment(Attachments.ASSERTED),
195                   LogMatchers.hasNotLogger(sysTest::isInExpectingErrorCategories),
196                   LogMatchers.hasLevel(Level.ERROR))));
197       ExpectLogs expectLogs = description.getAnnotation(ExpectLogs.class);
198       if (expectLogs != null) {
199         ExpectLog[] value = expectLogs.value();
200         for (ExpectLog expect : value) {
201           assertions.add(
202                   sysTest.expect(expect.category(), expect.level(), expect.nrTimes(),
203                           expect.expectationTimeout(), expect.timeUnit(),
204                   Matchers.allOf(LogMatchers.hasMessageWithPattern(expect.messageRegexp()),
205                   LogMatchers.hasLevel(expect.level()))));
206         }
207       } else {
208           ExpectLog expect = description.getAnnotation(ExpectLog.class);
209           if (expect != null) {
210             assertions.add(
211                   sysTest.expect(expect.category(), expect.level(), expect.nrTimes(),
212                   expect.expectationTimeout(), expect.timeUnit(),
213                   Matchers.allOf(LogMatchers.hasMessageWithPattern(expect.messageRegexp()),
214                   LogMatchers.hasLevel(expect.level()))));
215           }
216       }
217       return assertions;
218     }

220   private LogCollection<ArrayDeque<TestLogRecord>> handleLogCollections(final Description description,
221           final TestLoggers sysTest) {
222     CollectLogs ca = description.getAnnotation(CollectLogs.class);
223     Level mll;
224     boolean clp;
225     int nrLogs;
226     String include;
227     String[] exclude;
228     if (ca == null) {
229       mll = minLogLevel;
230       clp = collectPrinted;
231       nrLogs = maxDebugLogsCollected;
232       include = "";
233       exclude = excludeLogsFromCollection;
234     } else {
235       mll = ca.minLevel();
236       clp = ca.collectPrinted();
237       nrLogs = ca.nrLogs();
238       include = ca.includeLogs();
239       exclude = ca.excludeLogs();
240     }
241     return sysTest.collect(mll, nrLogs, clp, include, exclude);
242   }
243 
244   private void handlePrintLogAnnotations(final Description description, final TestLoggers sysTest) {
245     PrintLogsConfigs prtAnnots = description.getAnnotation(PrintLogsConfigs.class);
246     if (prtAnnots != null) {
247       for (PrintLogs prtAnnot : prtAnnots.value()) {
248         sysTest.print(prtAnnot.category(), TestLoggers.EXECUTED_FROM_IDE
249                 ? prtAnnot.ideMinLevel() : prtAnnot.minLevel(), prtAnnot.greedy());
250       }
251     } else {
252       PrintLogs prtAnnot = description.getAnnotation(PrintLogs.class);
253       if (prtAnnot != null) {
254         sysTest.print(prtAnnot.category(), TestLoggers.EXECUTED_FROM_IDE
255                 ? prtAnnot.ideMinLevel() : prtAnnot.minLevel(), prtAnnot.greedy());
256       }
257     }
258   }

261   @Override
262   @SuppressFBWarnings("AFBR_ABNORMAL_FINALLY_BLOCK_RETURN")
263   public synchronized void testFinished(final Description description) {
264     TestTimeSource.clear();
265     TestBaggage baggage = baggages.remove(description);
266     try (LogCollection<ArrayDeque<TestLogRecord>> h = baggage.getLogCollection()) {
267       handleUncaughtExceptions(description, h.get());
268     } finally {
269       try {
270         for (LogAssert assertion : baggage.getAssertions()) {
271           try {
272             assertion.assertObservation();
273           } catch (AssertionError ae) {
274             throw new AssertionError("Failed test " + description + ", " + ae.getMessage(), ae);
275           }
276         }
277       } finally {
278         closeAllContextCloseables(baggage);
279       }
280     }
281   }
282 
283   private void closeAllContextCloseables(final TestBaggage baggage) {
284     ExecutionContext ctx = baggage.getCtx();
285     ExecutionContext currentThreadContext = ExecutionContexts.current();
286     if (ctx == currentThreadContext) {
287       ctx.close();
288     } else {
289       throw new IllegalStateException("JUnit Threading model not as expected " + ctx + " != "
290               + currentThreadContext);
291     }
292   }

294   @Override
295   public synchronized void testFailure(final Failure failure) {
296     Description description = failure.getDescription();
297     TestBaggage bg = baggages.get(description);
298     if (bg != null) { // will Happen when a Uncaught Exception causes a test to fail.
299       LogCollection<ArrayDeque<TestLogRecord>> handler = bg.getLogCollection();
300       try (LogCollection<ArrayDeque<TestLogRecord>> h = handler) {
301         dumpDebugInfo(h.get(), description);
302       }
303     }
304   }
305 
306   private void handleUncaughtExceptions(final Description description,
307           @WillNotClose final Collection<TestLogRecord> logs) {
308     List<UncaughtExceptionDetail> exceptions = uncaughtExceptionHandler.getUncaughtExceptions();
309     if (!exceptions.isEmpty()) {
310       AssertionError assertionError = new AssertionError("Uncaught exceptions encountered " + exceptions);
311       for (UncaughtExceptionDetail ex : exceptions) {
312         Throwable throwable = ex.getThrowable();
313         LOG.info("Uncaught exceptions during {} in thread {}", description, ex.getThread(), ex.getThrowable());
314         assertionError.addSuppressed(throwable);
315       }
316       dumpDebugInfo(logs, description);
317       throw assertionError;
318     }
319   }
