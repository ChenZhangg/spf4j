65  /**
66   * Utility to sample stack traces. Stack traces can be persisted for later analysis.
67   *
68   * please read http://sape.inf.usi.ch/sites/default/files/publication/pldi10.pdf pure java stack sampling will probably
69   * have safepoint bias.

71   * @author zoly

146   public Sampler(final int sampleTimeMillis, final int dumpTimeMillis, final SamplerSupplier collector,
147           final File dumpFolder, final String dumpFilePrefix) {
148     CharSequences.validatedFileName(dumpFilePrefix);
149     stopped = true;
150     if (sampleTimeMillis < 1) {
151       throw new IllegalArgumentException("Invalid sample time " + sampleTimeMillis);
152     }
153     this.sampleTimeNanos = TimeUnit.MILLISECONDS.toNanos(sampleTimeMillis);
154     if (sampleTimeNanos < 0) {
155       throw new IllegalArgumentException("Invalid sample time " + sampleTimeMillis);
156     }
157     this.dumpTimeNanos = TimeUnit.MILLISECONDS.toNanos(dumpTimeMillis);
158     this.stackCollectorSupp = collector;
159     this.filePrefix = dumpFilePrefix;
160     this.dumpFolder = dumpFolder;
161   }

163   public static synchronized Sampler getSampler(final int sampleTimeMillis,
164           final int dumpTimeMillis,
165           final File dumpFolder, final String dumpFilePrefix) throws InterruptedException {
166     return getSampler(sampleTimeMillis, dumpTimeMillis,
167             (t) -> new FastStackCollector(false, true, new Thread[]{t}), dumpFolder, dumpFilePrefix);
168   }
169 
170   public static synchronized Sampler getSampler(final int sampleTimeMillis,
171           final int dumpTimeMillis, final SamplerSupplier collector,
172           final File dumpFolder, final String dumpFilePrefix) throws InterruptedException {
173     if (instance == null) {
174       try {
175         instance = new Sampler(sampleTimeMillis, dumpTimeMillis, collector,
176                 dumpFolder.getCanonicalFile(), dumpFilePrefix);
177       } catch (IOException ex) {
178         throw new UncheckedIOException(ex);
179       }
180       instance.registerJmx();
181       return instance;
182     } else {
183       instance.dispose();
184       try {
185         instance = new Sampler(sampleTimeMillis, dumpTimeMillis, collector,
186                 dumpFolder.getCanonicalFile(), dumpFilePrefix);
187       } catch (IOException ex) {
188         throw new UncheckedIOException(ex);
189       }
190       instance.registerJmx();
191       return instance;
192     }
193   }

199   @JmxExport(description = "start stack sampling")
200   public void start() {
201     synchronized (sync) {
202       if (stopped) {
203         stopped = false;
204         final long stNanos = sampleTimeNanos;
205         samplerFuture = DefaultExecutor.INSTANCE.submit(new AbstractRunnable("SPF4J-Sampling-Thread") {
206 
207           @SuppressWarnings("SleepWhileInLoop")
208           @SuppressFBWarnings({"MDM_THREAD_YIELD", "PREDICTABLE_RANDOM"})
209           @Override
210           public void doRun() {
211             lastDumpTimeNanos = TimeSource.nanoTime();
212             synchronized (sync) {
213               stackCollector = stackCollectorSupp.get(Thread.currentThread());
214             }
215             final long lDumpTimeNanos = dumpTimeNanos;
216             final ThreadLocalRandom random = ThreadLocalRandom.current();
217             long dumpCounterNanos = 0;
218             long sleepTimeNanos = 0;
219             long halfStNanos = stNanos / 2;
220             if (halfStNanos == 0) {
221               halfStNanos = 1;
222             }
223             long maxSleeepNanos = stNanos + halfStNanos;
224             while (true) {
225               try {
226                 synchronized (sync) {
227                   stackCollector.sample();
228                   if (stopped) {
229                     break;
230                   }
231                 }
232                 dumpCounterNanos += sleepTimeNanos;
233                 if (dumpCounterNanos >= lDumpTimeNanos) {
234                   long nanosSinceLastDump = TimeSource.nanoTime() - lastDumpTimeNanos;
235                   if (nanosSinceLastDump >= lDumpTimeNanos) {
236                     dumpCounterNanos = 0;
237                     File dumpFile = dumpToFile();
238                     if (dumpFile != null) {
239                       Logger.getLogger(Sampler.class.getName())
240                               .log(Level.INFO, "Stack samples written to {0}", dumpFile);
241                     }
242                   } else {
243                     dumpCounterNanos = nanosSinceLastDump;
244                   }
245                 }
246                 sleepTimeNanos = random.nextLong(halfStNanos, maxSleeepNanos);
247                 TimeUnit.NANOSECONDS.sleep(sleepTimeNanos);
248               } catch (InterruptedException ex) {
249                 Thread.currentThread().interrupt();
250                 return;
251               } catch (IOException | RuntimeException ex) {
252                 Logger.getLogger(Sampler.class.getName()).log(Level.SEVERE,
253                         "Exception encountered while samplig, will continue sampling", ex);
254               }
255             }
256           }
257         });
258       } else {
259         throw new IllegalStateException("Sampling can only be started once for " + this);
260       }
261     }
262   }

270   /**
271    * Dumps the sampled stacks to file. the collected samples are reset
272    *
273    * @param id - id will be added to file name returns the name of the file.
274    * @return - the file name where the data was persisted or null if there was no data to persist.
275    * @throws IOException - io issues while persisting data.
276    */
277   @JmxExport(value = "dumpToSpecificFile", description = "save stack samples to file")
278   @Nullable
279   @SuppressFBWarnings("PATH_TRAVERSAL_IN") // not possible the provided ID is validated for path separators.
280   public File dumpToFile(
281           @JmxExport(value = "fileID", description = "the ID that will be part of the file name")
282           @Nullable final String id) throws IOException {
283     String fileName = filePrefix + CharSequences.validatedFileName(((id == null) ? "" : '_' + id) + '_'
284             + DateTimeFormats.COMPACT_TS_FORMAT.format(
285                     Timing.getCurrentTiming().fromNanoTimeToInstant(lastDumpTimeNanos))
286             + '_' + DateTimeFormats.COMPACT_TS_FORMAT.format(Instant.now()));
287     File file = new File(dumpFolder, fileName);
288     return dumpToFile(file);
289   }
290 
291   @Nullable
292   @SuppressFBWarnings("PATH_TRAVERSAL_IN") // not possible the provided ID is validated for path separators.
293   public File dumpToFile(@Nonnull final File pfile) throws IOException {
294     Map<String, SampleNode> collections;
295     synchronized (sync) {
296       if (stackCollector == null) {
297         return null;
298       }
299       collections = stackCollector.getCollectionsAndReset();
300       lastDumpTimeNanos = TimeSource.nanoTime();
301     }
302     if (collections.isEmpty()) {
303       return null;
304     }
305     if (collections.size() == 1) {
306       Map.Entry<String, SampleNode> es = collections.entrySet().iterator().next();
307       SampleNode samples = es.getValue();
308       if (samples == null) {
309         return null;
310       }
311       File file;
312       if (pfile.getName().endsWith(".ssdump2")) {
313         file = pfile;
314       } else {
315         file = new File(pfile.getParentFile(), pfile.getName() + '_' + es.getKey() + ".ssdump2");
316       }
317       Converter.save(file, samples);
318       return file;
319     } else {
320       File file;
321       if (pfile.getName().endsWith(".ssdump3")) {
322         file = pfile;
323       } else {
324         file = new File(pfile.getParentFile(), pfile.getName() + ".ssdump3");
325       }
326       Converter.saveLabeledDumps(file, collections);
327       return file;
328     }
329   }

331   @JmxExport(description = "stop stack sampling")
332   public void stop() throws InterruptedException {
333     Future<?> toCancel = null;
334     synchronized (sync) {
335       if (!stopped) {
336         stopped = true;
337         toCancel = samplerFuture;
338       }
339     }
340     if (toCancel != null) {
341       try {
342         toCancel.get(dumpTimeNanos * 3, TimeUnit.NANOSECONDS);
343       } catch (TimeoutException ex) {
344         toCancel.cancel(true);
345         throw new Spf4jProfilerException(ex);
346       } catch (ExecutionException ex) {
347         throw new Spf4jProfilerException(ex);
348       }
349     }
350   }
