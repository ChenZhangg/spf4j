112   private void attachMavenId(final Schema schema) {
113     if (schema.getProp("mvnId") == null) {
114       StringBuilder idBuilder = new StringBuilder(64);
115       idBuilder.append(mavenProject.getGroupId()).append(':').append(mavenProject.getArtifactId())
116               .append(':').append(mavenProject.getVersion());
117       StringBuilder idb = new StringBuilder(4);
118       AppendableUtils.appendUnsignedString(idb, idSequence, 5);
119       idBuilder.append(':').append(idb);
120       schema.addProp("mvnId", idBuilder.toString());
121       index.put(idb.toString(), schema);
122       idSequence++;
123     }
124   }

126   protected void doCompileIDL(final String filename) throws IOException {
127     try {
128       List<String> cpElements = mavenProject.getCompileClasspathElements();
129       Idl parser;
130 
131       List<URL> runtimeUrls = new ArrayList<URL>();
132 
133       // Add the source directory of avro files to the classpath so that
134       // imports can refer to other idl files as classpath resources
135       runtimeUrls.add(sourceDirectory.toURI().toURL());
136       runtimeUrls.add(dependenciesDirectory.toURI().toURL());
137 
138       // If runtimeClasspathElements is not empty values add its values to Idl path.
139       if (cpElements != null && !cpElements.isEmpty()) {
140         for (Object runtimeClasspathElement : cpElements) {
141           String element = (String) runtimeClasspathElement;
142           runtimeUrls.add(new File(element).toURI().toURL());
143         }
144       }
145       getLog().info("Compile classpath: " + runtimeUrls);
146       URLClassLoader projPathLoader = AccessController.doPrivileged(
147               (PrivilegedAction<URLClassLoader>) ()
148                       -> new URLClassLoader(runtimeUrls.toArray(new URL[runtimeUrls.size()]),
149               Thread.currentThread().getContextClassLoader()));
150       File file = new File(sourceDirectory, filename);
151       parser = new Idl(file, projPathLoader);
152       Protocol protocol = parser.CompilationUnit();
153       Collection<Schema> types = protocol.getTypes();
154       for (Schema schema : types) {
155         if (addMavenId) {
156           attachMavenId(schema);
157         }
158         String targetName = schema.getFullName().replaceAll("\\.", File.separator) + ".avsc";
159         Path destinationFile = generatedAvscTarget.toPath().resolve(targetName);
160         Path parent = destinationFile.getParent();
161         if (parent != null) {
162           Files.createDirectories(parent);
163         }
164         Files.write(destinationFile,
165                 schema.toString().getBytes(StandardCharsets.UTF_8),
166                 StandardOpenOption.CREATE);
167       }
168       SpecificCompiler compiler = new SpecificCompiler(protocol);
169       compiler.setStringType(GenericData.StringType.String);
170       compiler.setTemplateDir(templateDirectory);
171       compiler.setFieldVisibility(SpecificCompiler.FieldVisibility.valueOf(fieldVisibility));
172       compiler.setCreateSetters(createSetters);
173       compiler.compileToDestination(null, generatedJavaTarget);
174     } catch (ParseException e) {
175       throw new IOException(e);
176     } catch (DependencyResolutionRequiredException drre) {
177       throw new IOException(drre);
178     }
179   }

181   protected void doCompileSchemas(final String[] filenames)
182           throws IOException {
183     Schema.Parser parser = new Schema.Parser();
184     for (String fileName : filenames) {
185       File src = new File(sourceDirectory, fileName);
186       Schema schema = parser.parse(src);
187       if (addMavenId) {
188         attachMavenId(schema);
189       }
190       String targetName = schema.getFullName().replaceAll("\\.", File.separator) + ".avsc";
191       Path destination = generatedAvscTarget.toPath().resolve(targetName);
192       Path parent = destination.getParent();
193       if (parent != null) {
194         Files.createDirectories(parent);
195       }
196       Files.write(destination,
197               schema.toString().getBytes(StandardCharsets.UTF_8),
198               StandardOpenOption.CREATE);
199       SpecificCompiler compiler = new SpecificCompiler(schema);
200       compiler.setTemplateDir(templateDirectory);
201       compiler.setStringType(GenericData.StringType.String);
202       compiler.setFieldVisibility(SpecificCompiler.FieldVisibility.valueOf(fieldVisibility));
203       compiler.setCreateSetters(createSetters);
204       compiler.setOutputCharacterEncoding(mavenProject.getProperties().getProperty("project.build.sourceEncoding"));
205       compiler.compileToDestination(src, generatedJavaTarget);
206     }
207   }

234   public void deleteGeneratedAvailableInDependencies() throws IOException {
235     Path classesInfo = dependenciesDirectory.toPath().resolve("classes.txt");
236     Set<String> classes = new HashSet(Files.readAllLines(classesInfo, StandardCharsets.UTF_8));
237     Path javaPath = generatedJavaTarget.toPath();
238     List<Path> dupes = Files.walk(javaPath)
239             .filter((p) -> {
240               Path relativize = javaPath.relativize(p);
241               return classes.contains(relativize.toString().replace(".java", ".class"));
242             }).collect(Collectors.toList());
243     for (Path p : dupes) {
244       Files.delete(p);
245     }
246     getLog().info("Deleted dupes: " + dupes);
247   }

249   public void deleteProtocolClasses() throws IOException {
250    String detectionString = "org.apache.avro.Protocol PROTOCOL";
251     Path javaPath = generatedJavaTarget.toPath();
252     List<Path> protocolFiles = Files.walk(javaPath)
253             .filter((p) -> {
254               Path fileName = p.getFileName();
255               if (fileName == null || !fileName.toString().endsWith(".java")) {
256                 return false;
257               }
258               try (BufferedReader br = Files.newBufferedReader(p, StandardCharsets.UTF_8)) {
259                 String line;
260                 while ((line = br.readLine()) != null) {
261                   if (line.contains(detectionString)) {
262                     return true;
263                   }
264                 }
265               } catch (IOException ex) {
266                 throw new UncheckedIOException(ex);
267               }
268               return false;
269             }).collect(Collectors.toList());
270     for (Path p : protocolFiles) {
271       Files.delete(p);
272     }
273   }

275   public void deleteSchemasAvailableInDependencies(final Path schTargetPath) throws IOException {
276     Path classesInfo = dependenciesDirectory.toPath();
277     Set<Path> schemas = Files.walk(classesInfo).filter(
278             (p) -> {
279               Path fileName = p.getFileName();
280               return fileName == null ? false : fileName.toString().endsWith("avsc");
281             })
282             .map((p) -> classesInfo.relativize(p)).collect(Collectors.toSet());
283     List<Path> dupes = Files.walk(schTargetPath).filter((p) -> schemas.contains(schTargetPath.relativize(p)))
284             .collect(Collectors.toList());
285     for (Path p : dupes) {
286       Files.delete(p);
287     }
288     getLog().info("Deleted dupes: " + dupes);
289   }

294   @Override
295   public void execute() throws MojoExecutionException, MojoFailureException {
296     Log logger = this.getLog();
297     logger.info("Generationg java code + schemas");
298     synchronized (String.class) {
299       for (Map.Entry<String, String> entry : ((Set<Map.Entry<String, String>>) ((Set) systemProperties.entrySet()))) {
300         System.setProperty(entry.getKey(), entry.getValue());
301       }
302       Path pSources = this.target.toPath().resolve("avro-sources");
303       String[] sourceFiles = getSourceFiles("**/*.avsc");
304       try {
305         doCompileSchemas(sourceFiles);
306       } catch (IOException ex) {
307         throw new MojoExecutionException("cannot compile schemas " + Arrays.toString(sourceFiles), ex);
308       }
309 
310       for (String file : getSourceFiles("**/*.avpr")) {
311         try {
312           doCompileProtocol(file);
313           Path destination = pSources.resolve(file);
314           Path folder = destination.getParent();
315           if (folder != null) {
316             Files.createDirectories(folder);
317           }
318           Files.copy(sourceDirectory.toPath().resolve(file), destination, StandardCopyOption.REPLACE_EXISTING);
319         } catch (IOException ex) {
320           throw new MojoExecutionException("cannot compile protocol " + file, ex);
321         }
322       }
323       for (String file : getSourceFiles("**/*.avdl")) {
324         try {
325           doCompileIDL(file);
326           Path destination = pSources.resolve(file);
327           Path parent = destination.getParent();
328           if (parent != null) {
329             Files.createDirectories(parent);
330           }
331           Files.copy(sourceDirectory.toPath().resolve(file), destination, StandardCopyOption.REPLACE_EXISTING);
332         } catch (IOException ex) {
333           throw new MojoExecutionException("cannot compile IDL " + file, ex);
334         }
335       }
336       Path codegenManifest = generatedAvscTarget.toPath().resolve(SCHEMA_MANIFEST);
337       try {
338         Files.write(codegenManifest,
339              Collections.singletonList("Build-Time=" + DateTimeFormatter.ISO_INSTANT.format(Instant.now()) + '\n'),
340              StandardCharsets.UTF_8);
341       } catch (IOException ex) {
342         throw new MojoExecutionException("Cannot create codegen manifest file " + codegenManifest, ex);
343       }
344       try {
345         deleteGeneratedAvailableInDependencies();
346         deleteSchemasAvailableInDependencies(getGeneratedAvscTarget().toPath());
347         if (deleteProtocolInterface) {
348           deleteProtocolClasses();
349         }
350       } catch (IOException ex) {
351         throw new MojoExecutionException("Cannot delete dependency dupes " + this, ex);
352       }
353       Path indexFile = this.generatedAvscTarget.toPath().resolve("schema_index.properties");
354       try (BufferedWriter bw = Files.newBufferedWriter(indexFile,
355               StandardCharsets.UTF_8, StandardOpenOption.CREATE, StandardOpenOption.TRUNCATE_EXISTING)) {
356         for (Map.Entry<String, Schema> entry : index.entrySet()) {
357             bw.append(entry.getKey());
358             bw.append('=');
359             bw.append(entry.getValue().getFullName());
360             bw.append('\n');
361         }
362       } catch (IOException ex) {
363         throw new MojoExecutionException("Cannot generate schema index " + this, ex);
364       }
365       mavenProject.addCompileSourceRoot(generatedJavaTarget.getAbsolutePath());
366       Resource resource = new Resource();
367       resource.setDirectory(this.generatedAvscTarget.getAbsolutePath());
368       resource.addInclude("**/*.avsc");
369       resource.addInclude("*.properties");
370       mavenProject.addResource(resource);
371       Resource resource2 = new Resource();
372       resource2.setDirectory(pSources.toString());
373       resource2.addInclude("**/*.avpr");
374       resource2.addInclude("**/*.avdl");
375       mavenProject.addResource(resource2);
376     }
377   }

379   public String[] getSourceFiles(final String pattern) {
380     FileSetManager fsm = new FileSetManager();
381     FileSet fs = new FileSet();
382     fs.setDirectory(sourceDirectory.getAbsolutePath());
383     fs.addInclude(pattern);
384     fs.setFollowSymlinks(false);
385     return fsm.getIncludedFiles(fs);
386   }
