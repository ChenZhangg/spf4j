51  import java.io.IOException;
52  import java.io.Writer;
53  
54  
55  public final class OctalUnescaper extends CharSequenceTranslator {
56  
57      /**
58       * {@inheritDoc}
59       */
60      @Override
61      public int translate(final CharSequence input, final int index, final Writer out) throws IOException {
62          int remaining = input.length() - index - 1; // how many characters left, ignoring the first \
63          if (remaining > 0 && input.charAt(index) == '\\' && isOctalDigit(input.charAt(index + 1))) {
64              StringBuilder builder = new StringBuilder();
65              int next = index + 1;
66              int next2 = index + 2;
67              int next3 = index + 3;
68  
69              // we know this is good as we checked it in the if block above
70              builder.append(input.charAt(next));
71  
72              if (remaining > 1 && isOctalDigit(input.charAt(next2))) {
73                  builder.append(input.charAt(next2));
74                  if (remaining > 2 && isZeroToThree(input.charAt(next)) && isOctalDigit(input.charAt(next3))) {
75                      builder.append(input.charAt(next3));
76                  }
77              }
78  
79              out.write(Integer.parseInt(builder.toString(), 8));
80              return 1 + builder.length();
81          }
82          return 0;
83      }
84  
85      /**
86       * Checks if the given char is an octal digit. Octal digits are the character representations of the digits 0 to 7.
87       * @param ch the char to check
88       * @return true if the given char is the character representation of one of the digits from 0 to 7
89       */
90      private boolean isOctalDigit(final char ch) {
91          return ch >= '0' && ch <= '7';
92      }
93  
94      /**
95       * Checks if the given char is the character representation of one of the digit from 0 to 3.
96       * @param ch the char to check
97       * @return true if the given char is the character representation of one of the digits from 0 to 3
98       */
99      private boolean isZeroToThree(final char ch) {
100         return ch >= '0' && ch <= '3';
101     }
102 }
