117   private static volatile Predicate<Throwable> nonRecoverableClassificationPredicate = new Predicate<Throwable>() {
118     @Override
119     @SuppressFBWarnings("ITC_INHERITANCE_TYPE_CHECKING")
120     public boolean test(final Throwable t) {
121 
122       if (t instanceof Error && !(t instanceof StackOverflowError)
123               && !(t.getClass().getName().endsWith("TokenMgrError"))) {
124         return true;
125       }
126       if (t instanceof IOException) {
127         String message = t.getMessage();
128         if (message != null && message.contains("Too many open files")) {
129           return true;
130         }
131       }
132       return false;
133     }
134   };

137   private static volatile Predicate<Throwable> isRetryablePredicate = new Predicate<Throwable>() {
138 
139     /**
140      * A default predicate that will return true if a exception is retriable...
141      * @param t
142      * @return
143      */
144     @Override
145     @SuppressFBWarnings("ITC_INHERITANCE_TYPE_CHECKING")
146     public boolean test(final Throwable t) {
147       // non recoverables are not retryable.
148       if (Throwables.containsNonRecoverable(t)) {
149         return false;
150       }
151       // Root Cause
152       Throwable rootCause = com.google.common.base.Throwables.getRootCause(t);
153       if (rootCause instanceof SSLException) {
154         return false;
155       }
156       if (rootCause instanceof RuntimeException) {
157         String name = rootCause.getClass().getName();
158         if (name.contains("NonTransient") || !name.contains("Transient")) {
159           return false;
160         }
161       }
162       // check causal chaing
163       Throwable e = Throwables.firstCause(t,
164               (ex) -> {
165                 String exClassName = ex.getClass().getName();
166                 return (ex instanceof SQLTransientException
167               || ex instanceof SQLRecoverableException
168               || (ex instanceof IOException && !exClassName.contains("Json"))
169               || ex instanceof TimeoutException
170               || (exClassName.contains("Transient")
171                         && !exClassName.contains("NonTransient")));
172                         });
173       return e != null;
174     }
175   };

202     try {
203       final List<Throwable> suppressedExceptions = (List<Throwable>) SUPPRESSED_FIELD.get(t);
204       if (suppressedExceptions != null) {
205         return suppressedExceptions.size();
206       } else {

210       throw new RuntimeException(ex);
211     }

214   public static int getNrRecursiveSuppressedExceptions(final Throwable t) {
215     try {
216       final List<Throwable> suppressedExceptions = (List<Throwable>) SUPPRESSED_FIELD.get(t);
217       if (suppressedExceptions != null) {
218         int count = 0;
219         for (Throwable se : suppressedExceptions) {
220           count += 1 + getNrRecursiveSuppressedExceptions(se);
221         }
222         return count;
223       } else {
224         return 0;
225       }
226     } catch (IllegalArgumentException | IllegalAccessException ex) {
227       throw new RuntimeException(ex);
228     }
229   }

231   @Nullable
232   public static Throwable removeOldestSuppressedRecursive(final Throwable t) {
233     try {
234       final List<Throwable> suppressedExceptions = (List<Throwable>) SUPPRESSED_FIELD.get(t);
235       if (suppressedExceptions != null && !suppressedExceptions.isEmpty()) {
236         Throwable ex = suppressedExceptions.get(0);
237         if (getNrSuppressedExceptions(ex) > 0) {
238           return removeOldestSuppressedRecursive(ex);
239         } else {
240           return suppressedExceptions.remove(0);
241         }
242       } else {
243         return null;
244       }
245     } catch (IllegalArgumentException | IllegalAccessException ex) {
246       throw new RuntimeException(ex);
247     }
248   }

313   public static <T extends Throwable> T chain(final T t, final Throwable newRootCause, final int maxChained) {
314     int chainedExNr = com.google.common.base.Throwables.getCausalChain(t).size();
315     if (chainedExNr >= maxChained) {
316       T res = clone0(t);
317       res.addSuppressed(new TrimmedException("Max chained excetions exceeded " + maxChained));
318       return res;
319     }
320     List<Throwable> newRootCauseChain = com.google.common.base.Throwables.getCausalChain(newRootCause);
321     int newChainIdx = 0;
322     final int size = newRootCauseChain.size();
323     if (chainedExNr + size > maxChained) {
324       newChainIdx = size - (maxChained - chainedExNr);
325       t.addSuppressed(new TrimmedException("Trimming exception at " + newChainIdx));
326     }
327     T result = clone0(t);
328     chain0(result, newRootCauseChain.get(newChainIdx));
329     return result;
330 
331   }

333   private static <T extends Throwable> T clone0(final T t) {
334     T result;
335     try {
336       result = Objects.clone(t);
337     } catch (RuntimeException ex) {
338       result = t;
339       t.addSuppressed(new TrimmedException("Unable to clone exception " + t));
340     }
341     return result;
342   }

344   public static void trimCausalChain(final Throwable t, final int maxSize) {
345     List<Throwable> causalChain = com.google.common.base.Throwables.getCausalChain(t);
346     if (causalChain.size() <= maxSize) {
347       return;
348     }
349     setCause(causalChain.get(maxSize - 1), null);
350   }

367   @CheckReturnValue
368   public static <T extends Throwable> T suppress(@Nonnull final T t, @Nonnull final Throwable suppressed,
369           final int maxSuppressed) {
370     T clone;
371     try {
372       clone = Objects.clone(t);
373     } catch (RuntimeException ex) {
374       t.addSuppressed(ex);
375       clone = t;
376     }
377     clone.addSuppressed(suppressed);
378     while (getNrRecursiveSuppressedExceptions(clone) > maxSuppressed) {
379       if (removeOldestSuppressedRecursive(clone) == null) {
380         throw new IllegalArgumentException("Impossible state for " + clone);
381       }
382     }
383     return clone;
384   }

386   /**
387    * Utility to get suppressed exceptions.
388    *
389    * In java 1.7 it will return t.getSuppressed() + in case it is Iterable<Throwable> any other linked exceptions (see
390    * SQLException)
391    *
392    * java 1.6 behavior is deprecated.
393    *
394    * @param t
395    * @return
396    */
397   public static Throwable[] getSuppressed(final Throwable t) {
398     if (t instanceof Iterable) {
399       // see SQLException
400       List<Throwable> suppressed = new ArrayList<>(java.util.Arrays.asList(t.getSuppressed()));
401       Set<Throwable> ignore = new HashSet<>();
402       ignore.addAll(com.google.common.base.Throwables.getCausalChain(t));
403       Iterator it = ((Iterable) t).iterator();
404       while (it.hasNext()) {
405         Object next = it.next();
406         if (next instanceof Throwable) {
407           if (ignore.contains((Throwable) next)) {
408             continue;
409           }
410           suppressed.add((Throwable) next);
411           ignore.addAll(com.google.common.base.Throwables.getCausalChain((Throwable) next));
412         } else {
413           break;
414         }
415       }
416       return suppressed.toArray(new Throwable[suppressed.size()]);
417     } else {
418       return t.getSuppressed();
419     }
420 
421   }

423   public static void writeTo(final StackTraceElement element, @Nullable final StackTraceElement previous,
424           final Appendable to, final PackageDetail detail,
425           final boolean abbreviatedTraceElement)
426           throws IOException {
427     String currClassName = element.getClassName();
428     String prevClassName = previous == null ? null : previous.getClassName();
429     if (abbreviatedTraceElement) {
430       if (currClassName.equals(prevClassName)) {
431         to.append('^');
432       } else {
433         writeAbreviatedClassName(currClassName, to);
434       }
435     } else {
436       to.append(currClassName);
437     }
438     to.append('.');
439     to.append(element.getMethodName());
440     String currFileName = element.getFileName();
441     String fileName = currFileName;
442     if (abbreviatedTraceElement && java.util.Objects.equals(currFileName,
443             previous == null ? null : previous.getFileName())) {
444       fileName = "^";
445     }
446     final int lineNumber = element.getLineNumber();
447     if (element.isNativeMethod()) {
448       to.append("(Native Method)");
449     } else if (fileName != null && lineNumber >= 0) {
450       to.append('(').append(fileName).append(':')
451               .append(Integer.toString(lineNumber)).append(')');
452     } else if (fileName != null) {
453       to.append('(').append(fileName).append(')');
454     } else {
455       to.append("(Unknown Source)");
456     }
457     if (detail == PackageDetail.NONE) {
458       return;
459     }
460     if (abbreviatedTraceElement && currClassName.equals(prevClassName)) {
461       to.append("[^]");
462       return;
463     }
464     PackageInfo pInfo = PackageInfo.getPackageInfo(currClassName);
465     if (abbreviatedTraceElement && prevClassName != null && pInfo.equals(PackageInfo.getPackageInfo(prevClassName))) {
466       to.append("[^]");
467       return;
468     }
469     if (pInfo.hasInfo()) {
470       String jarSourceUrl = pInfo.getUrl();
471       String version = pInfo.getVersion();
472       to.append('[');
473       if (jarSourceUrl != null) {
474         if (detail == PackageDetail.SHORT) {
475           String url = jarSourceUrl;
476           int lastIndexOf = url.lastIndexOf('/');
477           if (lastIndexOf >= 0) {
478             int lpos = url.length() - 1;
479             if (lastIndexOf == lpos) {
480               int prevSlPos = url.lastIndexOf('/', lpos - 1);
481               if (prevSlPos < 0) {
482                 to.append(url);
483               } else {
484                 to.append(url, prevSlPos + 1, url.length());
485               }
486             } else {
487               to.append(url, lastIndexOf + 1, url.length());
488             }
489           } else {
490             to.append(url);
491           }
492         } else {
493           to.append(jarSourceUrl);
494         }
495       } else {
496         to.append("na");
497       }
498       if (version != null) {
499         to.append(':');
500         to.append(version);
501       }
502       to.append(']');
503     }
504   }

574   public static void writeTo(final Throwable t, final Appendable to, final PackageDetail detail,
575           final boolean abbreviatedTraceElement) throws IOException {
576 
577     Set<Throwable> dejaVu = Collections.newSetFromMap(new IdentityHashMap<Throwable, Boolean>());
578     dejaVu.add(t);
579 
580     writeMessageString(to, t);
581     to.append('\n');
582     StackTraceElement[] trace = t.getStackTrace();
583 
584     writeTo(trace, to, detail, abbreviatedTraceElement);
585 
586     // Print suppressed exceptions, if any
587     for (Throwable se : getSuppressed(t)) {
588       printEnclosedStackTrace(se, to, trace, SUPPRESSED_CAPTION, "\t", dejaVu, detail, abbreviatedTraceElement);
589     }
590 
591     Throwable ourCause = t.getCause();
592 
593     // Print cause, if any
594     if (ourCause != null) {
595       printEnclosedStackTrace(ourCause, to, trace, CAUSE_CAPTION, "", dejaVu, detail, abbreviatedTraceElement);
596     }
597

621   public static int commonFrames(final StackTraceElement[] trace, final StackTraceElement[] enclosingTrace) {
622     int m = trace.length - 1;
623     int n = enclosingTrace.length - 1;
624     while (m >= 0 && n >= 0 && trace[m].equals(enclosingTrace[n])) {
625       m--;
626       n--;
627     }
628     return trace.length - 1 - m;
629   }

631   private static void printEnclosedStackTrace(final Throwable t, final Appendable s,
632           final StackTraceElement[] enclosingTrace,
633           final String caption,
634           final String prefix,
635           final Set<Throwable> dejaVu,
636           final PackageDetail detail,
637           final boolean abbreviatedTraceElement) throws IOException {
638     if (dejaVu.contains(t)) {
639       s.append("\t[CIRCULAR REFERENCE:");
640       writeMessageString(s, t);
641       s.append(']');
642     } else {
643       dejaVu.add(t);
644       // Compute number of frames in common between this and enclosing trace
645       StackTraceElement[] trace = t.getStackTrace();
646       int framesInCommon = commonFrames(trace, enclosingTrace);
647       int m = trace.length - framesInCommon;
648       // Print our stack trace
649       s.append(prefix).append(caption);
650       writeMessageString(s, t);
651       s.append('\n');
652       StackTraceElement prev = null;
653       for (int i = 0; i < m; i++) {
654         s.append(prefix).append("\tat ");
655         StackTraceElement ste = trace[i];
656         writeTo(ste, prev, s, detail, abbreviatedTraceElement);
657         s.append('\n');
658         prev = ste;
659       }
660       if (framesInCommon != 0) {
661         s.append(prefix).append("\t... ").append(Integer.toString(framesInCommon)).append(" more");
662         s.append('\n');
663       }
664 
665       // Print suppressed exceptions, if any
666       for (Throwable se : getSuppressed(t)) {
667         printEnclosedStackTrace(se, s, trace, SUPPRESSED_CAPTION, prefix + '\t',
668                 dejaVu, detail, abbreviatedTraceElement);
669       }
670 
671       // Print cause, if any
672       Throwable ourCause = t.getCause();
673       if (ourCause != null) {
674         printEnclosedStackTrace(ourCause, s, trace, CAUSE_CAPTION, prefix,
675                 dejaVu, detail, abbreviatedTraceElement);
676       }
677     }
678   }

712    * @param <T>
713    * @param t
714    * @param clasz
715    * @return
716    */
717   @Nullable
718   @CheckReturnValue
719   public static <T extends Throwable> T first(@Nonnull final Throwable t, final Class<T> clasz) {
720     return (T) first(t, (Throwable th) -> clasz.isAssignableFrom(th.getClass()));
721   }

723   /**
724    * Returns the first Throwable that matches the predicate in the causal and suppressed chain.
725    * @param t the Throwable
726    * @param predicate the Predicate
727    * @return the Throwable the first matches the predicate or null is none matches.
728    */
729   @Nullable
730   @CheckReturnValue
731   public static Throwable first(@Nonnull final Throwable t, final Predicate<Throwable> predicate) {
732     ArrayDeque<Throwable> toScan =  new ArrayDeque<>();
733     toScan.addFirst(t);
734     Throwable th;
735     THashSet<Throwable> seen = new IdentityHashSet<>();
736     while ((th = toScan.pollFirst()) != null) {
737       if (seen.contains(th)) {
738         continue;
739       }
740       if (predicate.test(th)) {
741         return th;
742       } else {
743         Throwable cause = th.getCause();
744         if (cause != null) {
745           toScan.addFirst(cause);
746         }
747         for (Throwable supp : th.getSuppressed()) {
748           toScan.addLast(supp);
749         }
750       }
751       seen.add(th);
752     }
753     return null;
754   }

757   /**
758    * Returns first Throwable in the causality chain that is matching the provided predicate.
759    * @param throwable the Throwable to go through.
760    * @param predicate the predicate to apply
761    * @return the first Throwable from the chain that the predicate matches.
762    */
763 
764   @Nullable
765   @CheckReturnValue
766   public static Throwable firstCause(@Nonnull final Throwable throwable, final Predicate<Throwable> predicate) {
767     Throwable t = throwable;
768     do {
769       if (predicate.test(t)) {
770         return t;
771       }
772       t = t.getCause();
773     } while (t != null);
774     return null;
775   }

790   public static void writeAbreviatedClassName(final String className, final Appendable writeTo) throws IOException {
791     int ldIdx = className.lastIndexOf('.');
792     if (ldIdx < 0) {
793       writeTo.append(className);
794       return;
795     }
796     boolean isPreviousDot = true;
797     for (int i = 0; i < ldIdx; i++) {
798       char c = className.charAt(i);
799       boolean isCurrentCharDot = c == '.';
800       if (isPreviousDot || isCurrentCharDot) {
801         writeTo.append(c);
802       }
803       isPreviousDot = isCurrentCharDot;
804     }
805     writeTo.append(className, ldIdx, className.length());
806   }

808   @SuppressFBWarnings("ITC_INHERITANCE_TYPE_CHECKING")
809   public static void throwException(final Exception ex) throws IOException, InterruptedException,
810           ExecutionException, TimeoutException {
811     if (ex instanceof IOException) {
812       throw (IOException) ex;
813     } else if (ex instanceof InterruptedException) {
814       throw (InterruptedException) ex;
815     } else if (ex instanceof ExecutionException) {
816       throw (ExecutionException) ex;
817     } else if (ex instanceof TimeoutException) {
818       throw (TimeoutException) ex;
819     } else {
820       throw new ExecutionException(ex);
821     }
822   }
