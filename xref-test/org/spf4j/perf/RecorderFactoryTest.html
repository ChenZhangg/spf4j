55    /**
56     * Test of createScalableQuantizedRecorder method, of class RecorderFactory.
57     */
58    @Test
59    public void testCreateScalableQuantizedRecorder() throws  IOException, InterruptedException {
60      String forWhat = "test1";
61      String unitOfMeasurement = "ms";
62      int sampleTime = 1000;
63      int factor = 10;
64      int lowerMagnitude = 0;
65      int higherMagnitude = 3;
66      int quantasPerMagnitude = 10;
67      CloseableMeasurementRecorder result = RecorderFactory.createScalableQuantizedRecorder2(
68              forWhat, unitOfMeasurement, sampleTime, factor, lowerMagnitude, higherMagnitude, quantasPerMagnitude);
69      for (int i = 0; i < 500; i++) {
70        result.record(i);
71        Thread.sleep(20);
72      }
73      result.close();
74      assertData(forWhat, 124750);
75    }

82      Object forWhat = "bla";
83      String unitOfMeasurement = "ms";

122   @Test
123   public void testCreateScalableCountingRecorderSource() throws IOException, InterruptedException {
124     String forWhat = "counters";
125     String unitOfMeasurement = "counts";
126     int sampleTime = 1000;
127     CloseableMeasurementRecorderSource result = RecorderFactory.createScalableCountingRecorderSource2(
128             forWhat, unitOfMeasurement, sampleTime);
129     for (int i = 0; i < 5000; i++) {
130       result.getRecorder("X" + i % 2).record(1);
131       Thread.sleep(1);
132     }
133     result.close();
134     assertData(forWhat + ",X1", 2500);
135   }

137   @SuppressFBWarnings("CLI_CONSTANT_LIST_INDEX")
138   public static void assertData(final String forWhat, final long expectedValue) throws IOException {
139     TSDBWriter dbWriter = ((TSDBMeasurementStore) RecorderFactory.MEASUREMENT_STORE).getDBWriter();
140     dbWriter.flush();
141     final File file = dbWriter.getFile();
142     List<TableDef> tableDefs = TSDBQuery.getTableDef(file, forWhat);
143     TableDef tableDef = tableDefs.get(0);
144     TimeSeries timeSeries = TSDBQuery.getTimeSeries(file, new long[]{tableDef.getId()}, 0, Long.MAX_VALUE);
145     long sum = 0;
146     long[][] values = timeSeries.getValues();
147     for (long[] row : values) {
148       sum += row[0];
149     }
150     Assert.assertEquals(expectedValue, sum);
151   }
