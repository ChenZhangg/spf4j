52  @SuppressFBWarnings("FCCD_FIND_CLASS_CIRCULAR_DEPENDENCY")
53  public final class Spf4jTestLogRunListenerSingleton extends RunListener {
54  
55    private static final Logger LOG = LoggerFactory.getLogger(Spf4jTestLogRunListenerSingleton.class);
56  
57    private static final Spf4jTestLogRunListenerSingleton INSTANCE = new Spf4jTestLogRunListenerSingleton();
58  
59    static {
60      System.setProperty("spf4j.timeSource", TestTimeSource.class.getName());
61    }
62  
63    private final Level minLogLevel;
64  
65    private final int maxDebugLogsCollected;
66  
67    private final Map<Description, LogCollection<ArrayDeque<LogRecord>>> collections;
68  
69    private final Map<Description, ExecutionContext> ctxts;
70  
71    private final boolean collectPrinted;
72  
73    private final ExceptionAsserterUncaughtExceptionHandler uncaughtExceptionHandler;
74  
75    private Spf4jTestLogRunListenerSingleton() {
76      minLogLevel = Level.valueOf(System.getProperty("spf4j.test.log.collectMinLevel", "DEBUG"));
77      maxDebugLogsCollected = Integer.getInteger("spf4j.test.log.collectmaxLogs", 100);
78      collectPrinted = Boolean.getBoolean("spf4j.test.log.collectPrintedLogs");
79      collections = new ConcurrentHashMap<>();
80      ctxts = new ConcurrentHashMap<>();
81      synchronized (Thread.class) {
82        final Thread.UncaughtExceptionHandler defaultHandler = Thread.getDefaultUncaughtExceptionHandler();
83        uncaughtExceptionHandler = new ExceptionAsserterUncaughtExceptionHandler(defaultHandler);
84        Thread.setDefaultUncaughtExceptionHandler(uncaughtExceptionHandler);
85      }
86    }
87  
88    public static Spf4jTestLogRunListenerSingleton getInstance() {
89      return INSTANCE;
90    }
91  
92    @Override
93    public void testRunFinished(final Result result) {
94      List<UncaughtExceptionDetail> exceptions = uncaughtExceptionHandler.getUncaughtExceptions();
95      if (!exceptions.isEmpty()) {
96        AssertionError assertionError = new AssertionError("Uncaught exceptions encountered " + exceptions);
97        for (UncaughtExceptionDetail ex : exceptions) {
98          Throwable throwable = ex.getThrowable();
99          assertionError.addSuppressed(throwable);
100         LOG.info("Uncaught exceptions, failures = {} in thread {}", result.getFailures(),
101                 ex.getThread(), throwable);
102       }
103       throw assertionError;
104     }
105   }
106 
107   @Override
108   public void testFailure(final Failure failure) {
109     Description description = failure.getDescription();
110     LogCollection<ArrayDeque<LogRecord>> handler = collections.get(description);
111     if (handler != null) { // will Happen when a Uncaught Exception causes a test to fail.
112       try (LogCollection<ArrayDeque<LogRecord>> h = handler) {
113         dumpDebugInfo(h.get(), description);
114       }
115     }
116   }
117 
118   private void dumpDebugInfo(final Collection<LogRecord> logs,
119           final Description description) {
120     synchronized (System.out) { // do not interleave other stuff.
121       boolean first = true;
122       if (!logs.isEmpty()) {
123         for (LogRecord record : logs) {
124           if (first) {
125             LOG.info("Dumping last {} logs collected for debug for {}", maxDebugLogsCollected, description);
126             first = false;
127           }
128           LogPrinter.printTo(System.out, record, "");
129         }
130         LOG.info("End dump for {}", description);
131       }
132     }
133 
134   }
135 
136   @Override
137   @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
138   public void testStarted(final Description description) throws Exception {
139     Test ta = description.getAnnotation(Test.class);
140     ExecutionContext ctx;
141     if (ta != null && ta.timeout() > 0) {
142       ctx = ExecutionContexts.start(description.getDisplayName(), ta.timeout(), TimeUnit.MILLISECONDS);
143     } else {
144       ctx = ExecutionContexts.start(description.getDisplayName());
145     }
146     CollectLogs ca = description.getAnnotation(CollectLogs.class);
147     Level mll = ca == null ? minLogLevel : ca.minLevel();
148     boolean clp = ca == null ? collectPrinted : ca.collectPrinted();
149     TestLoggers sysTest = TestLoggers.sys();
150     collections.put(description, sysTest.collect(mll, maxDebugLogsCollected, clp));
151     PrintLogsConfigs prtAnnots = description.getAnnotation(PrintLogsConfigs.class);
152     if (prtAnnots != null) {
153       PrintLogs[] value = prtAnnots.value();
154       for (PrintLogs prtAnnot : value) {
155         sysTest.print(prtAnnot.category(), TestLoggers.EXECUTED_FROM_IDE
156               ? prtAnnot.ideMinLevel() : prtAnnot.minLevel(), prtAnnot.greedy());
157       }
158     } else {
159       PrintLogs prtAnnot = description.getAnnotation(PrintLogs.class);
160       if (prtAnnot != null) {
161         sysTest.print(prtAnnot.category(), TestLoggers.EXECUTED_FROM_IDE
162               ? prtAnnot.ideMinLevel() : prtAnnot.minLevel(), prtAnnot.greedy());
163       }
164     }
165     ctxts.put(description, ctx);
166     super.testStarted(description);
167   }
168 
169 
170   @Override
171   public synchronized void testFinished(final Description description) {
172     TestTimeSource.clear();
173     LogCollection<ArrayDeque<LogRecord>> handler = collections.remove(description);
174     try (LogCollection<ArrayDeque<LogRecord>> h = handler) {
175       handleUncaughtExceptions(description, h.get());
176     }
177     ExecutionContext ctx = ctxts.remove(description);
178     ExecutionContext currentThreadContext = ExecutionContexts.current();
179     if (ctx == currentThreadContext) {
180       ctx.close();
181       List<AutoCloseable> closeables = (List<AutoCloseable>) ctx.get(AutoCloseable.class);
182       if (closeables != null) {
183         Exception ex = Closeables.closeAll(closeables);
184         if (ex != null) {
185           throw new IllegalStateException("cannot close " + closeables, ex);
186         }
187       }
188     } else {
189       throw new IllegalStateException("JUnit Threading model not as expected " + ctx + " != "
190               + currentThreadContext);
191     }
192   }
193 
194   private void handleUncaughtExceptions(final Description description,
195           @WillNotClose final Collection<LogRecord> logs) {
196     List<UncaughtExceptionDetail> exceptions = uncaughtExceptionHandler.getUncaughtExceptions();
197     if (!exceptions.isEmpty()) {
198       AssertionError assertionError = new AssertionError("Uncaught exceptions encountered " + exceptions);
199       for (UncaughtExceptionDetail ex : exceptions) {
200         Throwable throwable = ex.getThrowable();
201         LOG.info("Uncaught exceptions during {} in thread {}", description, ex.getThread(), ex.getThrowable());
202         assertionError.addSuppressed(throwable);
203       }
204       dumpDebugInfo(logs, description);
205       throw assertionError;
206     }
207   }
208 
209   public ExceptionHandoverRegistry getUncaughtExceptionHandler() {
210     return uncaughtExceptionHandler;
211   }
212 
213   @Override
214   public String toString() {
215     return "DetailOnFailureRunListener{" + "minLogLevel=" + minLogLevel
216             + ", maxDebugLogsCollected=" + maxDebugLogsCollected + ", collections="
217             + collections + ", collectPrinted=" + collectPrinted + '}';
218   }
