98    private Spf4jTestLogRunListenerSingleton() {
99      minLogLevel = Level.valueOf(System.getProperty("spf4j.test.log.collectMinLevel", "DEBUG"));
100     maxDebugLogsCollected = Integer.getInteger("spf4j.test.log.collectmaxLogs", 100);
101     defaultTestTimeoutMillis =  Long.getLong("spf4j.test.log.defaultTestTimeoutMillis", 120000);
102     collectPrinted = Boolean.getBoolean("spf4j.test.log.collectPrintedLogs");
103     baggages = new ConcurrentHashMap<>();
104     synchronized (Thread.class) {
105       final Thread.UncaughtExceptionHandler defaultHandler = Thread.getDefaultUncaughtExceptionHandler();
106       uncaughtExceptionHandler = new ExceptionAsserterUncaughtExceptionHandler(defaultHandler);
107       Thread.setDefaultUncaughtExceptionHandler(uncaughtExceptionHandler);
108     }

115   @Override
116   public void testRunFinished(final Result result) {
117     List<UncaughtExceptionDetail> exceptions = uncaughtExceptionHandler.getUncaughtExceptions();
118     if (!exceptions.isEmpty()) {
119       AssertionError assertionError = new AssertionError("Uncaught exceptions encountered " + exceptions);
120       for (UncaughtExceptionDetail ex : exceptions) {
121         Throwable throwable = ex.getThrowable();
122         assertionError.addSuppressed(throwable);
123         LOG.info("Uncaught exceptions, failures = {} in thread {}", result.getFailures(),
124                 ex.getThread(), throwable);
125       }
126       throw assertionError;
127     }
128   }

130   private void dumpDebugInfo(final Collection<TestLogRecord> logs,
131           final Description description) {
132     Iterator<TestLogRecord> iterator = logs.iterator();
133     if (iterator.hasNext()) {
134       synchronized (System.out) { // do not interleave other stuff.
135         LogPrinter.PRINTER.printTo(System.out, new TestLogRecordImpl("test", Level.INFO,
136                 "Dumping last {} logs collected for debug for {}", maxDebugLogsCollected, description), "");
137         TestLogRecord record = iterator.next();
138         LogPrinter.PRINTER.printTo(System.out, record, "");
139         while (iterator.hasNext()) {
140           record = iterator.next();
141           LogPrinter.PRINTER.printTo(System.out, record, "");
142         }
143         LogPrinter.PRINTER.printTo(System.out, new TestLogRecordImpl("test", Level.INFO,
144                 "End debug log dump for {}", description), "");
145       }
146     }

150   private static void registerDeadlockLogger(final Description description,
151           final ExecutionContext ctx, final long delay, final TimeUnit tu) {
152     final long delayMillis = Math.max(tu.toMillis(delay) - 10,  0);
153     ScheduledFuture<?> future = SCHEDULER.schedule(() -> {
154       LOG.info("Unit test  {} did not finish after {} {}, dumping thread stacks", description, delay, tu);
155       Threads.dumpToPrintStream(System.err);
156     }, delayMillis, TimeUnit.MILLISECONDS);
157     ctx.addCloseable(() -> {
158           future.cancel(true);
159         });
160   }

162   @Override
163   @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
164   public void testStarted(final Description description) throws Exception {
165     Test ta = description.getAnnotation(Test.class);
166     ExecutionContext ctx;
167     if (TestUtils.isExecutedWithDebuggerAgent()) {
168       ctx = ExecutionContexts.start(description.getDisplayName());
169     } else if (ta != null && ta.timeout() > 0) {
170       ctx = ExecutionContexts.start(description.getDisplayName(), ta.timeout(), TimeUnit.MILLISECONDS);
171       registerDeadlockLogger(description, ctx, ta.timeout(), TimeUnit.MILLISECONDS);
172     } else {
173       ctx = ExecutionContexts.start(description.getDisplayName(), defaultTestTimeoutMillis, TimeUnit.MILLISECONDS);
174       registerDeadlockLogger(description, ctx, defaultTestTimeoutMillis, TimeUnit.MILLISECONDS);
175     }
176     TestLoggers sysTest = TestLoggers.sys();
177     LogCollection<ArrayDeque<TestLogRecord>> collectLogs = handleLogCollections(description, sysTest);
178     List<LogAssert> logExpectations = setUpUnitTestLogExpectations(description, sysTest);
179     baggages.put(description, new TestBaggage(ctx, collectLogs, logExpectations));
180     handlePrintLogAnnotations(description, sysTest);
181     super.testStarted(description);
182   }
183 
184     private List<LogAssert> setUpUnitTestLogExpectations(final Description description,
185           final TestLoggers sysTest) {
186       List<LogAssert> assertions = new ArrayList<>(2);
187       assertions.add(
188                   sysTest.dontExpect("", Level.ERROR,
189                   Matchers.allOf(LogMatchers.noAttachment(Attachments.ASSERTED),
190                   LogMatchers.hasNotLoggers(sysTest.getExpectingErrorsIn()),
191                   LogMatchers.hasLevel(Level.ERROR))));
192       ExpectLogs expectLogs = description.getAnnotation(ExpectLogs.class);
193       if (expectLogs != null) {
194         ExpectLog[] value = expectLogs.value();
195         for (ExpectLog expect : value) {
196           assertions.add(
197                   sysTest.expect(expect.category(), expect.level(), expect.nrTimes(),
198                           expect.expectationTimeout(), expect.timeUnit(),
199                   Matchers.allOf(LogMatchers.hasMessageWithPattern(expect.messageRegexp()),
200                   LogMatchers.hasLevel(expect.level()))));
201         }
202       } else {
203           ExpectLog expect = description.getAnnotation(ExpectLog.class);
204           if (expect != null) {
205             assertions.add(
206                   sysTest.expect(expect.category(), expect.level(), expect.nrTimes(),
207                   expect.expectationTimeout(), expect.timeUnit(),
208                   Matchers.allOf(LogMatchers.hasMessageWithPattern(expect.messageRegexp()),
209                   LogMatchers.hasLevel(expect.level()))));
210           }
211       }
212       return assertions;
213     }

215   private LogCollection<ArrayDeque<TestLogRecord>> handleLogCollections(final Description description,
216           final TestLoggers sysTest) {
217     CollectLogs ca = description.getAnnotation(CollectLogs.class);
218     Level mll = ca == null ? minLogLevel : ca.minLevel();
219     boolean clp = ca == null ? collectPrinted : ca.collectPrinted();
220     return sysTest.collect(mll, maxDebugLogsCollected, clp);
221   }
222 
223   private void handlePrintLogAnnotations(final Description description, final TestLoggers sysTest) {
224     PrintLogsConfigs prtAnnots = description.getAnnotation(PrintLogsConfigs.class);
225     if (prtAnnots != null) {
226       for (PrintLogs prtAnnot : prtAnnots.value()) {
227         sysTest.print(prtAnnot.category(), TestLoggers.EXECUTED_FROM_IDE
228                 ? prtAnnot.ideMinLevel() : prtAnnot.minLevel(), prtAnnot.greedy());
229       }
230     } else {
231       PrintLogs prtAnnot = description.getAnnotation(PrintLogs.class);
232       if (prtAnnot != null) {
233         sysTest.print(prtAnnot.category(), TestLoggers.EXECUTED_FROM_IDE
234                 ? prtAnnot.ideMinLevel() : prtAnnot.minLevel(), prtAnnot.greedy());
235       }
236     }

240   @Override
241   @SuppressFBWarnings("AFBR_ABNORMAL_FINALLY_BLOCK_RETURN")
242   public synchronized void testFinished(final Description description) {
243     TestTimeSource.clear();
244     TestBaggage baggage = baggages.remove(description);
245     try (LogCollection<ArrayDeque<TestLogRecord>> h = baggage.getLogCollection()) {
246       handleUncaughtExceptions(description, h.get());
247     } finally {
248       try {
249         for (LogAssert assertion : baggage.getAssertions()) {
250           try {
251             assertion.assertObservation();
252           } catch (AssertionError ae) {
253             throw new AssertionError("Failed test " + description + ", " + ae.getMessage(), ae);
254           }
255         }
256       } finally {
257         closeAllContextCloseables(baggage);
258       }
259     }
260   }
261 
262   private  void closeAllContextCloseables(final TestBaggage baggage) {
263     ExecutionContext ctx = baggage.getCtx();
264     ExecutionContext currentThreadContext = ExecutionContexts.current();
265     if (ctx == currentThreadContext) {
266       ctx.close();
267     } else {
268       throw new IllegalStateException("JUnit Threading model not as expected " + ctx + " != "
269               + currentThreadContext);
270     }
271   }

273   @Override
274   public void testFailure(final Failure failure) {
275     Description description = failure.getDescription();
276     TestBaggage bg = baggages.get(description);
277     if (bg != null) { // will Happen when a Uncaught Exception causes a test to fail.
278       LogCollection<ArrayDeque<TestLogRecord>> handler = bg.getLogCollection();
279       try (LogCollection<ArrayDeque<TestLogRecord>> h = handler) {
280         dumpDebugInfo(h.get(), description);
281       }
282     }
283   }
284 
285   private void handleUncaughtExceptions(final Description description,
286           @WillNotClose final Collection<TestLogRecord> logs) {
287     List<UncaughtExceptionDetail> exceptions = uncaughtExceptionHandler.getUncaughtExceptions();
288     if (!exceptions.isEmpty()) {
289       AssertionError assertionError = new AssertionError("Uncaught exceptions encountered " + exceptions);
290       for (UncaughtExceptionDetail ex : exceptions) {
291         Throwable throwable = ex.getThrowable();
292         LOG.info("Uncaught exceptions during {} in thread {}", description, ex.getThread(), ex.getThrowable());
293         assertionError.addSuppressed(throwable);
294       }
295       dumpDebugInfo(logs, description);
296       throw assertionError;
297     }
298   }

301     return uncaughtExceptionHandler;
