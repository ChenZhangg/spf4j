67    public StackDumpJInternalFrame(final SampleNode samples,
68            final String title, final boolean isgraph) {
69      super(title);
70      setName(title);
71      initComponents();
72      if (samples == null) {
73        this.samples = SampleNode.createSampleNode(
74                new StackTraceElement[]{new StackTraceElement("NO SAMPLES", "", "", -1)});
75      } else {
76        this.samples = samples;
77      }
78      setViewType(isgraph);
79      ssScrollPanel.setVisible(true);
80      pack();
81    }

83    private void setViewType(final boolean isgraph) {
84      StackPanelBase view = (StackPanelBase) ssScrollPanel.getViewport().getView();
85      if (isgraph) {
86        graphToggle.setSelected(true);
87        //ssScrollPanel.setViewportView(new ZStackPanel(this.samples));
88        if (view != null) {
89          ssScrollPanel.setViewportView(new HotFlameStackPanel(view.getMethod(), view.getSamples(), view.getHistory()));
90        } else {
91          ssScrollPanel.setViewportView(new HotFlameStackPanel(Methods.ROOT, this.samples, new LinkedList<>()));
92        }
93      } else {
94        graphToggle.setSelected(false);
95        if (view != null) {
96          ssScrollPanel.setViewportView(new FlameStackPanel(view.getMethod(), view.getSamples(), view.getHistory()));
97        } else {
98          ssScrollPanel.setViewportView(new FlameStackPanel(Methods.ROOT, this.samples, new LinkedList<>()));
99        }
100     }
101   }

172   private void exportButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_exportButtonActionPerformed
173     JFileChooser fc = new JFileChooser();
174     FileFilter[] xFF = fc.getChoosableFileFilters();
175     for (FileFilter ff : xFF) {
176       fc.removeChoosableFileFilter(ff);
177     }
178     fc.addChoosableFileFilter(Spf4jFileFilter.D3_JSON);
179     fc.addChoosableFileFilter(Spf4jFileFilter.SPF4J_JSON);
180     fc.addChoosableFileFilter(Spf4jFileFilter.SSDUMP2);
181     fc.addChoosableFileFilter(Spf4jFileFilter.SSDUMP2_GZ);
182     fc.setFileFilter(Spf4jFileFilter.D3_JSON);
183     int retrival = fc.showSaveDialog(null);
184     if (retrival == JFileChooser.APPROVE_OPTION) {
185       File selectedFile = fc.getSelectedFile();
186       FileFilter fileFilter = fc.getFileFilter();
187       if (!(fileFilter instanceof Spf4jFileFilter)) {
188         throw new IllegalArgumentException("Invalid file type selected " + fileFilter);
189       }
190       String suffix = ((Spf4jFileFilter) fileFilter).getSuffix();
191       if (!selectedFile.getName().endsWith(suffix)) {
192         selectedFile = new File(selectedFile.getParentFile(), selectedFile.getName() + suffix);
193       }
194       if (Spf4jFileFilter.D3_JSON.accept(selectedFile)) { // D3 format.
195         try (BufferedWriter wr = Files.newBufferedWriter(selectedFile.toPath(), StandardCharsets.UTF_8)) {
196           samples.writeD3JsonTo(wr);
197         } catch (IOException ex) {
198           throw new UncheckedIOException(ex);
199         }
200       } else if (Spf4jFileFilter.SPF4J_JSON.accept(selectedFile)) {
201         try (BufferedWriter wr = Files.newBufferedWriter(selectedFile.toPath(), StandardCharsets.UTF_8)) {
202           samples.writeJsonTo(wr);
203         } catch (IOException ex) {
204           throw new UncheckedIOException(ex);
205         }
206       } else if (Spf4jFileFilter.SSDUMP2.accept(selectedFile) || Spf4jFileFilter.SSDUMP2_GZ.accept(selectedFile)) {
207         try {
208           Converter.save(selectedFile, samples);
209         } catch (IOException ex) {
210           throw new UncheckedIOException(ex);
211         }
212       } else {
213         throw new IllegalArgumentException("No ecognized extension for file " + selectedFile);
214       }
215     }
216   }//GEN-LAST:event_exportButtonActionPerformed
