34  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
35  import java.util.concurrent.Callable;
36  import java.util.concurrent.TimeUnit;
37  import java.util.concurrent.TimeoutException;
38  import javax.annotation.CheckReturnValue;
39  import org.spf4j.base.Either;
40  import org.spf4j.base.ExecutionContexts;
41  import org.spf4j.base.Throwables;
42  import org.spf4j.base.TimeSource;
43  import org.spf4j.base.UncheckedExecutionException;

45  /**
46   * @author Zoltan Farkas
47   */
48  public interface SyncRetryExecutor<T, C extends Callable<? extends T>> {
49  
50    @CheckReturnValue
51    <R extends T, W extends C, E extends Exception> R call(W pwhat, Class<E> exceptionClass,
52            long startNanos, long deadlineNanos)
53            throws InterruptedException, TimeoutException, E;
54  
55    @CheckReturnValue
56    default <R extends T, W extends C, E extends Exception> R call(W pwhat, Class<E> exceptionClass,
57            long deadlineNanos)
58            throws InterruptedException, TimeoutException, E {
59      return call(pwhat, exceptionClass, TimeSource.nanoTime(), deadlineNanos);
60    }
61  
62    @CheckReturnValue
63    default <R extends T, W extends C, E extends Exception> R call(W pwhat, Class<E> exceptionClass)
64            throws InterruptedException, TimeoutException, E {
65      long nanoTime = TimeSource.nanoTime();
66      return call(pwhat, exceptionClass, nanoTime, ExecutionContexts.getContextDeadlineNanos());
67    }
68  
69    @CheckReturnValue
70    default <R extends T, W extends C, E extends Exception> R call(W pwhat, Class<E> exceptionClass,
71            long timeout, TimeUnit tu)
72            throws InterruptedException, TimeoutException, E {
73      long nanoTime = TimeSource.nanoTime();
74      return call(pwhat, exceptionClass, nanoTime,
75              ExecutionContexts.computeDeadline(nanoTime, ExecutionContexts.current(), tu, timeout));
76    }
77  
78    default <W extends C, E extends Exception> void run(W pwhat, Class<E> exceptionClass)
79            throws InterruptedException, TimeoutException, E {
80      T res = call(pwhat, exceptionClass);
81      if (res != null) {
82        throw new IllegalStateException("result must be null not " + res);
83      }
84    }
85  
86    default <W extends C, E extends Exception> void run(W pwhat, Class<E> exceptionClass, long deadlineNanos)
87            throws InterruptedException, TimeoutException, E {
88      T res = call(pwhat, exceptionClass, deadlineNanos);
89      if (res != null) {
90        throw new IllegalStateException("result must be null not " + res);
91      }
92    }
93  
94    default <E extends Exception, W extends C> void run(W pwhat, Class<E> exceptionClass,
95            long timeout, TimeUnit tu)
96            throws InterruptedException, TimeoutException, E {
97       run(pwhat, exceptionClass, ExecutionContexts.computeDeadline(ExecutionContexts.current(), timeout, tu));
98    }

101   /**
102    * Naive implementation of execution with retry logic. a callable will be executed and retry attempted in current
103    * thread if the result and exception predicates. before retry, a callable can be executed that can abort the retry
104    * and finish the function with the previous result.
105    *
106    * @param <T> - The type of callable to retry result;
107    * @param <EX> - the exception thrown by the callable to retry.
108    * @param pwhat - the callable to retry.
109    * @return the result of the retried callable if successful.
110    * @throws java.lang.InterruptedException - thrown if retry interrupted.
111    * @throws EX - the exception thrown by callable.
112    */
113   @SuppressFBWarnings({ "MDM_THREAD_YIELD", "ITC_INHERITANCE_TYPE_CHECKING" })
114   static <T, E extends Exception, C extends Callable<? extends T>> T call(
115           final C pwhat,
116           final RetryPredicate<T, C> retryPredicate,
117           final Class<E> exceptionClass,
118           final int maxExceptionChain)
119           throws InterruptedException, TimeoutException, E {
120     C what = pwhat;
121     T result;
122     Throwable lastEx; // last exception
123     try {
124       result = what.call();
125       lastEx = null;
126     } catch (InterruptedException ex1) {
127       throw ex1;
128     } catch (Throwable e) { // only EX and RuntimeException
129       lastEx = e;
130       result = null;
131     }
132     Throwable lastExChain = lastEx; // last exception chained with all previous exceptions
133     RetryDecision<T, C> decision;
134     //CHECKSTYLE IGNORE InnerAssignment FOR NEXT 5 LINES
135     while ((lastEx != null)
136             ? (decision = retryPredicate.getExceptionDecision(lastEx, what)).getDecisionType()
137               == RetryDecision.Type.Retry
138             : (decision = retryPredicate.getDecision(result, what)).getDecisionType() == RetryDecision.Type.Retry) {
139       if (Thread.interrupted()) {
140         InterruptedException ex = new InterruptedException();
141         if (lastExChain != null) {
142           ex.addSuppressed(lastExChain);
143         }
144         throw ex;
145       }
146       long delayNanos = decision.getDelayNanos();
147       if (delayNanos > 0) {
148         TimeUnit.NANOSECONDS.sleep(delayNanos);
149       } else if (delayNanos < 0) {
150         throw new IllegalStateException("Invalid retry decision delay: " + delayNanos);
151       }
152       what = decision.getNewCallable();
153       try {
154         result = what.call();
155         lastEx = null;
156       } catch (InterruptedException ex1) {
157         if (lastExChain != null) {
158           ex1.addSuppressed(lastExChain);
159         }
160         throw ex1;
161       } catch (Exception e) { // only EX and RuntimeException
162         lastEx = e;
163         result = null;
164         if (lastExChain != null) {
165           Throwables.suppressLimited(lastEx, lastExChain, maxExceptionChain);
166         }
167         lastExChain = lastEx;
168       }
169     }
170     if (decision.getDecisionType() == RetryDecision.Type.Abort) {
171         Either<Throwable, T> r = decision.getResult();
172         if (r != null) {
173           if (r.isLeft()) {
174             lastEx = r.getLeft();
175             Throwable[] suppressed = lastEx.getSuppressed();
176             if (lastExChain != null && lastEx != lastExChain
177                     && (suppressed.length == 0 || suppressed[suppressed.length - 1] != lastExChain)
178                     && lastEx.getCause() != lastExChain) {
179               // we attach the chain in case the new exception does not.
180               Throwables.suppressLimited(lastEx, lastExChain, maxExceptionChain);
181             }
182             lastExChain = lastEx;
183           } else {
184             result = r.getRight();
185             lastEx = null;
186           }
187         }
188     } else {
189       throw new IllegalStateException("Should not happen, decision =  " + decision);
190     }
191     if (lastEx != null) {
192       if (lastExChain == null) {
193         lastExChain = lastEx;
194       }
195       if (lastExChain instanceof RuntimeException) {
196         throw (RuntimeException) lastExChain;
197       } else if (lastExChain instanceof TimeoutException) {
198         throw (TimeoutException) lastExChain;
199       } else if (exceptionClass.isAssignableFrom(lastExChain.getClass())) {
200         throw (E) lastExChain;
201       } else {
202         throw new UncheckedExecutionException(lastExChain);
203       }
204     }
205     return result;
206   }
